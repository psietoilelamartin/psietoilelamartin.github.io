On note $E=\mathscr{C}^0([0,1], \mathbb{R})$ que l'on munit des deux normes en posant:

$$
\|f\|_{\infty}=\sup\limits_{x \in[0,1]}(|f(x)|) \text { et }\|f\|_1=\displaystyle\int_0^1|f(t)| \mathrm{d} t
$$
On considère $A=\left\{f \in E, f(0)=0\right\}$ et $g: x \mapsto 1$.
\begin{enumerate}
\item Est-ce que $g$ est adhérent à $A$ pour la norme $\|\cdot\|_{\infty}$ ?
\item Est-ce que $g$ est adhérent à $A$ pour la norme $\|\cdot\|_1$ ?
\end{enumerate}
