Soient $E$ un espace euclidien et $f: E \rightarrow E$ une application linéaire vérifiant
$$
\forall\, x, y \in E,\ (x \mid y)=0 \Rightarrow(f(x) \mid f(y))=0
$$

\begin{enumerate}
\item Calculer $(u+v \mid u-v)$ pour $u, v$ vecteurs unitaires.
\item Établir qu'il existe $\alpha \in \mathbb{R}^{+}$ vérifiant $\forall\, x \in E$, $\|f(x)\|=\alpha\|x\|$.
\item Conclure qu'il existe $g \in \mathrm O(E)$ vérifiant $f=\alpha . g$
\end{enumerate}
