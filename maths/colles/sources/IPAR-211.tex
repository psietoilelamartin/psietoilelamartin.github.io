Soient $f(x)=\left(\displaystyle\int_0^x \mathrm{e}^{-t^2} \mathrm{d} t\right)^2$ et $g(x)=\displaystyle\int_0^1 \displaystyle\frac{\mathrm{e}^{-x^2\left(1+t^2\right)}}{1+t^2} \mathrm{d} t$.
\begin{enumerate}
\item Montrer que $f$ et $g$ sont de classe $\mathscr{C}^1$ sur $\mathbb{R}^{+}$et déterminer leur dérivée.
\item Montrer que pour tout $x \geqslant 0$, on a $f(x)+g(x)=\displaystyle\frac{\pi}{4}$.
\item En déduire $I=\displaystyle\int_0^{+\infty} \mathrm{e}^{-t^2} \mathrm{d} t$.
\end{enumerate}
