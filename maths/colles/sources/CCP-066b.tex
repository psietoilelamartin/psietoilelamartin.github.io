\begin{enumerate}
\item
Soit $A\in \cS_n\left( \mathbb{R}\right)$.\\
Prouver que $A\in \cS_n^+\left( \mathbb{R}\right)\Longleftrightarrow \Sp(A)\subset \left[0,+\infty \right[$.
	\:\:\:\:
\item
Prouver que $\forall\:A\in \cS_n\left( \mathbb{R}\right)$, $A^2\in \cS_n^+\left( \mathbb{R}\right)$.
	\:\:\:\:
\item

Prouver que $$\forall\:A\in \cS_n\left( \mathbb{R}\right),\ \forall\:B\in \cS_n^+\left( \mathbb{R}\right),\ AB=BA\ \Longrightarrow\ A^2B\in \cS_n^+\left( \mathbb{R}\right).$$
