Soit $A=(a_{i,j})_{1\leqslant i,j \leqslant n} \in \mathrm O_n(\mathbb{R})$. Montrer que
\setlength{\columnsep}{-0.8cm}
\begin{multicols}{3}
\begin{enumerate}
  \item $\abs{\dsum_{1\leqslant i,j\leqslant n} a_{i,j}} \leqslant n$
  \item $n \leqslant \dsum_{1\leqslant i,j\leqslant n} \abs{a_{i,j}} $
  \item $ \dsum_{1\leqslant i,j\leqslant n} \abs{a_{i,j}}  \leqslant n\sqrt{n}$
\end{enumerate}
\end{multicols}
\begin{enumerate}\setcounter{enumi}{3}
 \item Peut-on avoir simultanément $\sum_{1 \leqslant i, j \leqslant n}\left|a_{i, j}\right|=n \sqrt{n}$ et $\left|\sum_{1 \leqslant i, j \leqslant n} a_{i, j}\right|=n$ ?
\end{enumerate}
