Soit $A \in \mathrm{GL}_n(\mathbb{R})$.
\begin{enumerate}
\item Montrer que $A^\top A\in\mathscr{S}_n^{++}(\R)$.
\item Montrer que $A$ admet une \bemph{décomposition polaire} $A=\Omega S$
où $\Omega \in \mathrm{O}_n(\mathbb{R})$ et $S \in \mathscr{S}_n^{++}(\mathbb{R})$.\\
\emph{On pourra utiliser directement qu'il existe $S\in\mathscr{S}_n^{++}(\mathbb R)$ telle que $S^2=A^{\top}A$.}
\end{enumerate}
